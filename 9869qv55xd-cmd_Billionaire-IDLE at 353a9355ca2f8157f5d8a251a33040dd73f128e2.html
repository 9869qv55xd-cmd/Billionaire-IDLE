      <div class="section">
      <h3> Stock / Crypto</h3>
      <button onclick="invest()">High-Risk Invest</button>

      <h3> Prestige</h3>

      <script>
      let game = JSON.parse(localStorage.getItem("billionaire")) || {
      money:0,
      base:0,
      mult:1,
      prestige:1,
      owned:{},
      achievements:[],
      last:Date.now(),
      theme:"dark"
      };
      const data = {
      jobs:[{n:"Intern",c:50,i:1},{n:"Developer",c:500,i:10},{n:"CEO",c:50000,i:1000}],
      clothes:[{n:"Suit",c:500,m:1.2},{n:"Luxury Brand",c:5000,m:1.5}],
      cars:[{n:"Sports Car",c:10000,m:1.5},{n:"Supercar",c:100000,m:2}],
      homes:[{n:"Mansion",c:50000,m:2},{n:"Private Island",c:1000000,m:3}],
      employees:[{n:"Assistant",c:1000,i:5},{n:"Manager",c:10000,i:50}],
      companies:[{n:"Startup",c:50000,i:500},{n:"Corporation",c:500000,i:5000}]
      };
      function income(){ return game.base * game.mult * game.prestige; }
      function fmt(n){
      if(n>=1e9) return "$"+(n/1e9).toFixed(2)+"B";
      if(n>=1e6) return "$"+(n/1e6).toFixed(2)+"M";
      if(n>=1e3) return "$"+(n/1e3).toFixed(2)+"K";
      return "$"+Math.floor(n);
      }
      function tap(){ game.money += 1 * game.prestige; render(); }
      function buy(item){
      if(game.money<item.c||game.owned[item.n])return;
      game.money-=item.c;
      game.owned[item.n]=true;
      if(item.i) game.base+=item.i;
      if(item.m) game.mult*=item.m;
      render();
      }
      function draw(arr,id){
      let el=document.getElementById(id); el.innerHTML="";
      arr.forEach(i=>{
      let b=document.createElement("button");
      b.textContent=game.owned[i.n]?i.n+" ":`${i.n} (${fmt(i.c)})`;
      b.disabled=game.owned[i.n]||game.money<i.c;
      b.onclick=()=>buy(i);
      el.appendChild(b);
      });
      }
      function invest(){
      game.money *= Math.random()>0.5?1.4:0.7;
      render();
      }
      function doPrestige(){
      if(game.money<1e6) return alert("Need $1M");
      game.money=0; game.base=0; game.mult=1; game.owned={};
      game.prestige+=0.5;
      render();
      }
      function checkAchievements(){
      let list=document.getElementById("achievements");
      list.innerHTML="";
      [["Millionaire",1e6],["Billionaire",1e9]].forEach(a=>{
      if(game.money>=a[1]&&!game.achievements.includes(a[0]))
      game.achievements.push(a[0]);
      if(game.achievements.includes(a[0])){
      let li=document.createElement("li");
      li.textContent=" "+a[0];
      list.appendChild(li);
      }
      });
      }
      function toggleTheme(){
      game.theme = game.theme==="dark"?"light":"dark";
      document.body.className = game.theme==="light"?"light":"";
      }
      function offline(){
      let now=Date.now();
      game.money += ((now-game.last)/1000)*income();
      game.last=now;
      }
      function render(){
      document.getElementById("money").textContent=fmt(game.money);
      document.getElementById("income").textContent=fmt(income())+"/sec";
      document.getElementById("prestige").textContent="Prestige x"+game.prestige.toFixed(1);
      draw(data.jobs,"jobs");
      draw(data.clothes,"clothes");
      draw(data.cars,"cars");
      draw(data.homes,"homes");
      draw(data.employees,"employees");
      draw(data.companies,"companies");
      checkAchievements();
      localStorage.setItem("billionaire",JSON.stringify(game));
      }
      offline();
      setInterval(()=>{ game.money+=income(); render(); },1000);
      render();
      </script>
